public final class Point { private final int x, y; public Point() { x=0; y=0; } public Point(int x, int y) { this.x = x; this.y = y; } public int getX() { return x; } public int getY() { return y; } public String toString() { return "Point[" + "x=" + x + ", y=" + y + "]"; } public static void main(String[] args) { Point p = new Point (0, 0); System.out.println(p); Point p2 = new Point(10, 10); System.out.println(p2); } }
