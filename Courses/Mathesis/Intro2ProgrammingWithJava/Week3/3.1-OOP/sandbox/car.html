public class Car { // κλάση // ιδιότητες/γνωρίσματα String model; int maxSpeed; int ccm; int speed = 0; // μέθοδος δημιουργίας αντικειμένων - κατασκευαστής Car(String m, int s, int c) { model = m; maxSpeed = s; ccm = c; } // ενέργειες/μέθοδοι void accelerate() { if (speed <= maxSpeed - 10) speed+=10; } void decelerate() { if (speed >= 10) speed-=10; } public String toString() { return "Car[" + "model=" + model + ", maxSpeed=" + maxSpeed + ", ccm=" + ccm + ", speed=" + speed + "]"; } public static void main(String[] args) { Car audiA3 = new Car("Audi A3", 210, 1595); System.out.println(audiA3.speed); audiA3.accelerate(); System.out.println(audiA3.speed); } }
